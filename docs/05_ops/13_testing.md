## 13.2 単体テスト例

### GAS関数テスト

```javascript
// Test_JobService.gs
function test_saveJob_create() {
  const payload = {
    job_date: '2025-12-15',
    customer_id: 'test-customer-001',
    site_name: 'テスト現場',
    time_slot: 'AM',
    required_count: 2,
    status: 'OPEN'
  };

  const res = saveJob(payload, null);

  if (!res.ok) {
    throw new Error(`saveJob failed: [${res.error.code}] ${res.error.message} (${res.requestId})`);
  }
  if (!res.data?.job?.job_id) {
    throw new Error('job_id not generated');
  }

  Logger.log('test_saveJob_create: PASSED');
}

function test_getDashboard_byDate() {
  const res = getDashboard('2025-12-15');

  if (!res.ok) {
    throw new Error(`getDashboard failed: [${res.error.code}] ${res.error.message} (${res.requestId})`);
  }
  if (!Array.isArray(res.data?.jobs)) {
    throw new Error('data.jobs is not array');
  }

  Logger.log('test_getDashboard_byDate: PASSED');
}
```

## 13.6 性能テスト

### 測定項目

| 項目 | 目標値 |
|------|--------|
| ダッシュボード初期表示 | 3秒以内 |
| 案件一覧取得（50件） | 2秒以内 |
| 案件登録 | 1秒以内 |
| スタッフ検索（{{STAFF_MASTER_COUNT}}から） | 0.5秒以内 |

### 測定方法

```javascript
function performanceTest_getDashboard() {
  const start = new Date();

  const res = getDashboard('2025-12-15');

  const elapsed = new Date() - start;
  Logger.log(`getDashboard: ${elapsed}ms`);

  if (!res.ok) {
    throw new Error(`getDashboard failed: [${res.error.code}] ${res.error.message} (${res.requestId})`);
  }
  if (elapsed > 3000) {
    throw new Error('Performance threshold exceeded');
  }
}
```
